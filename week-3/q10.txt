#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;

    vector<int> arr(n);

    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    vector<int> prefix(n);

    prefix[0] = arr[0];

    for (int i = 1; i < n; i++) {
        prefix[i] = prefix[i - 1] ^ arr[i];
    }

    int q;
    cin >> q;

    for (int i = 0; i < q; i++) {
        int left, right;
        cin >> left >> right;

        int ans;

        if (left == 0)
            ans = prefix[right];
        else
            ans = prefix[right] ^ prefix[left - 1];

        cout << ans << " ";
    }

    return 0;
}
