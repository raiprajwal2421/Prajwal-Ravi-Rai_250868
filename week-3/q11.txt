#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, m;

    cin >> n;
    vector<int> nums1(n);

    for(int i = 0; i < n; i++) {
        cin >> nums1[i];
    }

    cin >> m;
    vector<int> nums2(m);

    for(int i = 0; i < m; i++) {
        cin >> nums2[i];
    }

    int ans = 0;

    if(m % 2 == 1) {
        for(int x : nums1) {
            ans ^= x;
        }
    }

    if(n % 2 == 1) {
        for(int x : nums2) {
            ans ^= x;
        }
    }

    cout << ans;

    return 0;
}
