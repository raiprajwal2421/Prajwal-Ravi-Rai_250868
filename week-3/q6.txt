#include <iostream>
#include <string>
#include <numeric>   // for gcd
using namespace std;

int main() {
    string str1, str2;

    cin >> str1 >> str2;

    if (str1 + str2 != str2 + str1) {
        cout << "";
        return 0;
    }

    int len = gcd(str1.length(), str2.length());

    cout << str1.substr(0, len);

    return 0;
}
