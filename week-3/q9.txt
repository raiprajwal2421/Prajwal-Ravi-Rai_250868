#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;

    vector<int> ans(n + 1);

    for (int i = 0; i <= n; i++) {
        int num = i;
        int count = 0;

        while (num > 0) {
            count += (num & 1);
            num >>= 1;
        }

        ans[i] = count;
    }

    for (int x : ans) {
        cout << x << " ";
    }

    return 0;
}
